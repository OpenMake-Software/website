/*

//OM      EXEC PGM=OOMLOM,

/*

//PARMS DD *

CFG=RELEASE

-f <enter Build Control file name>

-c <enter high level data set name for the build>

-a

-ov

-lo <enter user name to own logs>

-oh

-lj "<enter build job name>"

-lm <enter machine name>

-ld "<enter date for log file>"
